99 def get_T(pars):
100  # We always get the invariant tensor here, and cast it to the
101  # non-invariant if needed. This gets around the silly fact that the
102  # basis of the original tensor depends on symmetry_tensors,
103  # something that should be fixed.
104  T = tensordispenser.get_tensor(
105  pars, iter_count=0, symmetry_tensors=True)[0]
106  if not pars["symmetry_tensors"]:
107  T = Tensor.from_ndarray(T.to_ndarray())
108  log_fact = 0
109  Fs = []
110  Ns = []
111  cum = T
112  for i in range(1, pars["n_normalization"]):
113  cum = toolbox.contract2x2(cum)
114  log_fact *= 4
115  m = cum.abs().max()
116  if m != 0:
117  cum /= m
118  log_fact += np.log(m)
119  N = 4**i
120  F = np.log(scon(cum, [1,2,1,2]).value()) + log_fact
121  Fs.append(F)
122  Ns.append(N)
123  A, B = np.polyfit(Ns[pars["n_discard"]:], Fs[pars["n_discard"]:], 1)
124  T /= np.exp(A)
125  return T

128 def get_D(t, pars):
129  ham = (- pars["J"]*np.array([[ 1,-1],
130  [-1, 1]],
131  dtype=pars["dtype"])
132  + pars["H"]*np.array([[-1, 0],
133  [ 0, 1]],
134  dtype=pars["dtype"]))
135  boltz = np.exp(-pars["beta"]*ham)
136  ham_g = - pars["g"]* pars["J"]*np.array([[ 1,-1],
137  [-1, 1]],
138  dtype=pars["dtype"])
139  boltz_g = np.exp(-pars["beta"]*ham_g)
140  ones = np.ones((2,2), dtype=pars["dtype"])
141  D = np.einsum('ab,bc,cd,da->abcd', boltz, boltz_g, boltz_g, boltz)
142  u = np.array([[1, 1],
143  [1,-1]]) / np.sqrt(2)
144  u_dg = u.T.conjugate()
145  D = scon((D, u, u, u_dg, u_dg),
146  ([1,2,3,4], [-1,1], [-2,2], [3,-3], [4,-4]))
147  if pars["symmetry_tensors"]:
148  D = TensorZ2.from_ndarray(D, shape=[[1,1]]*4, qhape=[[0,1]]*4,
149  dirs=[1,1,-1,-1])
150  else:
151  D = Tensor.from_ndarray(D)
152  return D
