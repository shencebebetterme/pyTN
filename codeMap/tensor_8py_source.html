144  def find_trunc_dim(cls, S, chis=None, eps=0, break_degenerate=False,
145  degeneracy_eps=1e-6, norm_type="frobenius"):
146  S = np.abs(S)
147  if norm_type=="frobenius":
148  S = S**2
149  eps = eps**2
150  elif norm_type=="trace":
151  pass
152  else:
153  raise ValueError("Unknown norm_type {}".format(norm_type))
154  sum_all = sum(S)
155  # Find the smallest chi for which the error is small enough.
156  # If none is found, use the largest chi.
157  if sum_all != 0:
158  for chi in chis:
159  if not break_degenerate:
160  # Make sure that we don't break degenerate singular
161  # values by including one but not the other.
162  while 0 < chi < len(S):
163  last_in = S[chi-1]
164  last_out = S[chi]
165  rel_diff = np.abs(last_in-last_out)/last_in
166  if rel_diff < degeneracy_eps:
167  chi -= 1
168  else:
169  break
170  sum_disc = sum(S[chi:])
171  rel_err = sum_disc/sum_all
172  if rel_err <= eps:
173  break
174  if norm_type=="frobenius":
175  rel_err = np.sqrt(rel_err)
176  else:
177  rel_err = 0
178  chi = min(chis)
179  return chi, rel_err

184  def join_indices(self, *inds, return_transposed_shape_data=False,
185  **kwargs):
186  """ Joins indices together in the spirit of reshape. inds is
187  either a iterable of indices, in which case they are joined, or
188  a iterable of iterables of indices, in which case the indices
189  listed in each element of inds will be joined.
190 
191  Before any joining is done the indices are transposed so that
192  for every batch of indices to be joined the first remains in
193  place and the others are moved to be after in the order given.
194  The order in which the batches are given does not matter.
195 
196  If return_transposed_shape_data is True, then the shape of the
197  tensor after transposing but before reshaping is returned as
198  well, in addition to None and None, that take the place of
199  transposed_qhape and transposed_dirs of AbelianTensor.
200 
201  The method does not modify the original tensor.
202  """
203  # Format index_batches to be a list of lists of indices.
204  if isinstance(inds[0], collections.Iterable):
205  index_batches = list(map(list, inds))
206  else:
207  index_batches = [list(inds)]
208  # Remove empty batches.
209  index_batches = [b for b in index_batches if b]
210 
211  # Create the permutation p for transposing.
212  joined = set(sum(index_batches, []))
213  # Now we only need to insert the indices that are not joined.
214  not_joined = [[i] for i in range(len(self.shape)) if i not in joined]
215  all_in_batches = not_joined + index_batches
216  all_in_batches.sort(key=operator.itemgetter(0))
217  p = sum(all_in_batches, [])
218  index_batches = [batch for batch in all_in_batches if len(batch) > 1]
219  index_batches = [list(map(p.index, batch)) for batch in index_batches]
220  self = self.transpose(p)
221 
222  if return_transposed_shape_data:
223  transposed_shape = self.shape
224 
225  shp = list(self.shape)
226  for batch in reversed(index_batches):
227  new_dim = reduce(operator.mul, map(shp.__getitem__, batch))
228  shp[batch[0]] = new_dim
229  del shp[batch[1] : batch[0]+len(batch)]
230  self = self.reshape(shp)
231 
232  if return_transposed_shape_data:
233  return self, transposed_shape, None, None
234  else:
235  return self

238  def split_indices(self, indices, dims, qims=None, **kwargs):
239  """ Splits indices in the spirit of reshape. Indices is an
240  iterable of indices to be split. Dims is an iterable of
241  iterables such that dims[i]=dim_batch is an iterable of lists of
242  dimensions, each list giving the dimensions along a new index
243  that will come out of splitting indices[i].
244 
245  An example clarifies:
246  Suppose self has shape [dim1, dim2, dim3, dim4]. Suppose then
247  that indices = [1,3], dims = [[dimA, dimB], [dimC, dimD]]. Then
248  the resulting tensor will have shape [dim1, dimA, dimB, dim3,
249  dimC, dimD], assuming that that dims and are such that joining
250  dimA and dimB gives qim2, etc.
251 
252  In stead of a list of indices a single index may be given.
253  Correspondingly dims should then have one level of depth less as
254  well.
255 
256  split_indices never modifies the original tensor.
257  """
258  # Formatting the input so that indices is a list and dim_batches
259  # is a list of lists.
260  if isinstance(indices, collections.Iterable):
261  assert(len(indices) == len(dims))
262  indices = list(indices)
263  dim_batches = list(map(list, dims))
264  else:
265  indices = [indices]
266  dim_batches = [list(dims)]
267  dim_batches = [[type(self).flatten_dim(dim) for dim in batch]
268  for batch in dim_batches]
269 
270  if not indices:

274  # This is necessary for the next step to work.
