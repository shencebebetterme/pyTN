77  [1,-1] for matrices and [1] (respectively [-1]) if only
78  left_inds (respectively right_inds) is provided.

86  dirs = [1,-1]

277  def svd(self, a, b, chis=None, eps=0, print_errors=0,

279  norm_type="frobenius"):

281  other, and reshape self to a matrix. Then singular value
282  decompose this matrix into U, S, V. Finally reshape the unitary
283  matrices to tensors that have a new index coming from the SVD,
284  for U as the last index and for V as the first, and U to have
285  indices a as its first indices and V to have indices b as its
286  last indices.
287 
288  If eps>0 then the SVD may be truncated if the relative Frobenius
289  norm error can be kept below eps. For this purpose different
290  dimensions to truncate to can be tried, and these dimensions
291  should be listed in chis. If chis is None then the full range of
292  possible dimensions is tried.
293 
294  If print_errors > 0 then the truncation error is printed.
295 
296  If return_rel_err is True then the relative truncation error is
297  also returned.
298 
299  norm_type specifies the norm used to measure the error. This
300  defaults to "frobenius". The other option is "trace", for trace
301  norm.
302 
303  Output is U, S, V, and possibly rel_err. Here S is a vector of
304  singular values and U and V are isometric tensors (unitary if
305  the matrix that is SVDed is square and there is no truncation).
306  U . diag(S) . V = self, up to truncation errors.
307 
308  The original tensor is not modified by this method.

311  a = (a,)
312  if not isinstance(b, collections.Iterable):
313  b = (b,)

321  degeneracy_eps=degeneracy_eps,
322  norm_type=norm_type)

328  V_qims = (S.qhape, transposed_qhape[len(a):])

374  def split(self, a, b, chis=None, eps=0, print_errors=0,

378  """ Split with SVD. Like SVD, but takes the square root of the
379  singular values and multiplies both unitaries with it, so that
380  the tensor is split into two parts. Values are returned as
381  (US, {S}, SV, {rel_err}),
382  where the ones in curly brackets are only returned if the
383  corresponding arguments are True.

390  norm_type=norm_type)
391  U, S, V = svd_result[0:3]
392  S_sqrt = S.sqrt()
393  U = U.multiply_diag(S_sqrt, -1, direction="right")
394  V = V.multiply_diag(S_sqrt, 0, direction="left")
395  if return_sings:
