226 def generate_normalized_tensors(pars):
227  # Number of tensors to use to fix the normalization
228  n = max(8, pars["iter_count"] + 4)
229  # Number of tensors from the beginning to discard
230  n_discard = max(min(pars["iter_count"]-3, 3), 0)
231  tensors_and_log_facts = []
232  for i in reversed(list(range(n+1))):
233  tensors_and_log_facts.append(get_tensors(pars=pars, iter_count=i))
234  tensors_and_log_facts = tuple(reversed(tensors_and_log_facts))
235  A_lists = tuple(map(lambda t: t[0], tensors_and_log_facts))
236  log_fact_lists = np.array(tuple(map(lambda t: t[1], tensors_and_log_facts)))
237  Us = np.array(tuple(map(lambda t: t[2], tensors_and_log_facts)))
238 
239  Zs = np.array(tuple(scon(A_list, ([1,2,3,2], [3,4,1,4])).norm() for A_list in A_lists))
240  log_Zs = np.log(Zs)
241  log_Zs = np.array(tuple(log_Z + log_fact_list[0] + log_fact_list[1]
242  for log_Z, log_fact_list in zip(log_Zs, log_fact_lists)))
243  Ns = np.array([4*4**i-2**i for i in range(n+1)])
244  A, B = np.polyfit(Ns[pars["n_discard"]:], log_Zs[pars["n_discard"]:], 1)
245 
246  if pars["print_errors"]:
247  print("Fit when normalizing Ts: %.3e * N + %.3e"%(A,B))
248 
249  A_lists = [[A_list[0]/np.exp(N*A/2 - log_fact_list[0]),
250  A_list[1]/np.exp(N*A/2 - log_fact_list[1])]
251  for A_list, N, log_fact_list in zip(A_lists, Ns, log_fact_lists)]
252

255  write_tensor_file(data=(A_list, U), prefix="KW_tensors_normalized",
256  pars=id_pars, iter_count=i, filename=filename)
257

259  U = Us[pars["iter_count"]]

402  qimT = None

404  dim1 = T_first.shape[4]
405  try:

407  except TypeError:
408  qim1 = None
409  dim2 = T_first.shape[5]

411  qim2 = T_first.qhape[5]
412  except TypeError:

416  try:
417  qim1 = A1.qhape[3]
418  except TypeError:
419  qim1 = None
420  dim2 = A2.shape[3]
421  try:
422  qim2 = A2.qhape[3]
423  except TypeError:
424  qim2 = None
425  flatdimT = type(T).flatten_dim(dimT)
426  flatdim1 = type(T).flatten_dim(dim1)
427  flatdim2 = type(T).flatten_dim(dim2)
428  matrix_flatdim = flatdimT**(block_width-2)*flatdim1*flatdim2

430  def caster(v, charge=0):
431  v = type(T).from_ndarray(v, shape=[dim1, dim2] + [dimT]*(block_width-2),
432  qhape=[qim1, qim2] + [qimT]*(block_width-2),
433  charge=charge, dirs=[1]*block_width)
434  return v
435 
436  if do_coarse_momenta:
437  scon_list_end = [T_first] + [T]*(block_width-2)
438  index_list = [[i*2 + 1 for i in range(block_width)],
439  [block_width*2, -1, -2, 4, 1, 3]]
440  else:
441  scon_list_end = [A1, A2] + [T]*(block_width-2)
442  index_list = [[i*2 + 1 for i in range(block_width)],
443  [block_width*2, -1, 2, 1], [2, -2, 4, 3]]
444  for i in range(3, block_width+1):
445  index_list += [[2*i-2, -i, 2*i, 2*i-1]]
446 
447  if pars["do_momenta"] and not pars["do_coarse_momenta"]:
448  translation = [block_width-1] + list(range(block_width-1))
449 
450  def transfer_op(v, charge=0):
451  v = np.reshape(v, (flatdim1, flatdim2) + (flatdimT,)*(block_width-2))
452  v = caster(v, charge=charge)
453  if pars["do_momenta"] and not pars["do_coarse_momenta"]:
454  v = np.transpose(v, translation)
455  v = scon((U, v),
456  ([-1,-2,-3,1,2,3],
457  [1,2,3] + [-i for i in range(4, block_width+1)]))
458  print(".", end='', flush=True)
459  scon_list = [v] + scon_list_end
460  Av = scon(scon_list, index_list)
461  Av = Av.to_ndarray()
462  Av = np.reshape(Av, (matrix_flatdim,))
463  return Av
464 
465  print("Diagonalizing...", end="")
466  if hasattr(T, "sects"):
467  qnums = [0,1]
468  es = type(T).empty(shape=[[n_dims_do]*len(qnums)],
469  qhape=[qnums], invar=False,
470  dirs=[1], dtype=np.float_)
471  for q in qnums:

473  partial(transfer_op, charge=q),

477  tol=1e-8, return_eigenvectors=False)
478  else:

480  es_block = spsla.eigsh(transfer_op_lo, k=n_dims_do, maxiter=500,
481  tol=1e-8, return_eigenvectors=False)
482  order = np.argsort(-np.real(np.log(es_block)))
483  es_block = es_block[order]
484  es[(q,)] = es_block
485 
486  else:
487  transfer_op_lo = spsla.LinearOperator((matrix_flatdim, matrix_flatdim),
488  partial(transfer_op, charge=0),
489  dtype=T.dtype)
490  if pars["do_momenta"]:
491  es = spsla.eigs(transfer_op_lo, k=n_dims_do, maxiter=500,
492  tol=1e-8, return_eigenvectors=False)
493  else:
494  # Use the algorithm for Hermitian operators.
495  es = spsla.eigsh(transfer_op_lo, k=n_dims_do, maxiter=500,
496  tol=1e-8, return_eigenvectors=False)
497  es = type(T).from_ndarray(es)

499

501  log_es /= 2*np.pi
502  if do_coarse_momenta:
