368  Zs = np.array([scon(T, [1,2,1,2]).norm() for T in tensors])
369  cfg.string+="after all these RG steps, the norm of the final tensor A^{(s)} (self contraction) is equivalent to the partition function.<br/>"
370  log_Zs = np.log(Zs)
371  log_Zs += np.array(log_facts)
372 
373  if algo_name == "tnr":
374  Ns = np.array([2*4**i for i in range(n+1)])
375  elif algo_name == "trg":
