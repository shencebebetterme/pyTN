105 def get_defect(alpha, T, index):
106  # Build a group-like defect.
107  dim = T.shape[index]
108  try:
109  qim = T.qhape[index]
110  except TypeError:
111  qim = None
112  defect = type(T).eye(dim, qim=qim, dtype=T.dtype)
113  if alpha != 0:
114  for k,v in defect.sects.items():
115  phase = np.exp(1j*alpha*k[0])
116  defect[k] = v*phase
117  return defect

120 def get_T(pars):
121  # We always get the invariant tensor here, and cast it to the
122  # non-invariant one later if we so wish.
123  # This makes constructing the defects easier.
124  T = tensordispenser.get_tensor(pars, iter_count=0,
125  symmetry_tensors=True)[0]
126  log_fact = 0
127  Fs = []
128  Ns = []
129  cum = T
130  for i in range(1, pars["n_normalization"]):
131  cum = toolbox.contract2x2(cum)
132  log_fact *= 4
133  m = cum.abs().max()
134  if m != 0:
135  cum /= m
136  log_fact += np.log(m)
137  N = 4**i
138  F = np.log(scon(cum, [1,2,1,2]).value()) + log_fact
139  Fs.append(F)
140  Ns.append(N)
141  A, B = np.polyfit(Ns[pars["n_discard"]:], Fs[pars["n_discard"]:], 1)
142  T /= np.exp(A)
143  return T

160 def qnums_from_eigenvectors(evects, pars):
161  sites = len(evects.shape) - 1
162  if pars["model"].strip().lower() == "ising":
163  symop = np.array([[1,0], [0,-1]], dtype=np.float_)
164  symop = type(evects).from_ndarray(symop)
165  else:
166  # TODO generalize symop to models other than Ising.
167  NotImplementedError("Symmetry operators for models other than "
168  "Ising have not been implemented.")
169  ncon_list = (evects, evects.conjugate()) + (symop,)*sites
170  index_list = ((list(range(1,sites+1)) + [-1],
171  list(range(sites+1, 2*sites+1)) + [-2])
172  + tuple([i+sites,i] for i in range(1,sites+1)))
173  qnums = scon(ncon_list, index_list)
174  qnums = qnums.diag()
175  # Round to the closest possible qnum for the given model.
176  pos_qnums = initialtensors.symmetry_classes_dims_qims[pars["model"]][2]
177  max_qnum = max(pos_qnums)
178  qnums = qnums.astype(np.complex_).log()*(max_qnum+1)/(2j*np.pi)
179  qnums = [min(pos_qnums, key=lambda x: abs(x-q)) for q in qnums]
180  return qnums
