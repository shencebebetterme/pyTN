24 class AbelianTensor(TensorCommon):
25  """ A class for symmetry preserving tensors capabable of handling
26  all abelian symmetry groups. Meant to be subclassed to implement
27  specific symmetries, which can typically be done by simply fixing
28  the qodulus (see below) of the class.
29 
30  Every AbelianTensor has the following attributes:
31 
32  shape: A list of dims, one dim per leg. Every dim is a list of
33  integerers that are the dimensions of the different quantum number
34  blocks along that leg.
35 
36  qhape: A list of qims, one qim per leg. Every qim is a list of
37  unique integers that are the quantum numbers (qnums) of that leg.
38  The quantum numbers are in one-to-one correspondence with the
39  elements of the dims, so that qhape[i][j] and shape[i][j] are the
40  qnum and dimension of the same block.
41 
42  dirs: A list of integers -1 or 1, one for each leg. 1 means that the
43  corresponding leg is outgoing, -1 means incoming.
44 
45  qodulus: An integer or None. If an integer, then all arithmetic on
46  the quantum numbers is done modulo qodulus. If None then arithmetic
47  on qnums is just usual integer arithmetic.
48 
49  sects: A dict of numpy arrays, with combinations of quantum numbers
50  as keys. Every key must a tuple of quantum numbers, one for each
51  leg, and each one of them being from the qim of that leg. The value
52  of the dict at this key is the block (or "sector" or "sect")
53  corresponding to these quantum numbers. If the tensor is invariant
54  under a symmetry (see invar) then only certain blocks are allowed to
55  be set, but even in such a cause not all allowed blocks must be set.
56  For the treatement of unset blocks see defval.
57 
58  dtype: A numpy dtype, that is the dtype of all the sects.
59 
60  defval: The default value that the tensor has everywhere outside the
61  blocks set in sects. If the tensor is a scalar with no legs then its
62  value is its defval and it has no blocks. Note that many of the
63  methods - such as dot and svd - require defval == 0 (and assert
64  this). The main use of defval != 0 is to be able to handle tensors
65  of boolean values that arise in comparisons.
66 
67  charge: An integer such that if invar is True then all the blocks
68  set in sects must have keys k such that sum_i k[i]*dirs[i] % qodulus
69  == charge.
70 
71  invar: A boolean. If True, then the tensor is invariant under the
72  symmetry determined by qodulus, in the sense described in the
73  definition of charge. If False, this condition is ignored and any
74  block can be set. Note that as with defval, many methods require the
75  tensor to be invariant and invar == False is mainly meant for
76  handling vectors of singular values and eigenvalues. If invar ==
77  True then defval must be 0, unless the tensor is a scalar of charge
78  0.
79 
80  Note that many of these rules are not constantly checked for and can
81  be broken by the user. In such cases behavior of the class is not
82  quaranteed. The method check_consistency can be used to check that
83  the tensor conforms to this definition.
84 
85  """

437  def defer_binary_elementwise(self, B, op_func, *args, **kwargs):

441  form: same qnums, dims, etc. If not, either warnings or errors
442  are raised, depending on whether the operation can still be
443  carried out meaningfully.

446  the defval of self are operated on with op_func(_, B, *args,
447  **kwargs).
448

450  new tensor is like self in its attributes, but may be
451  non-invariant if its defval ends up being non-zero.
452

454  on tensors, such as basic arithmetic and comparisons.

464  # scalar. They should also have the same qodulus.

643  def to_ndarray(self):
644  """ Returns a corresponding numpy array. The order of the blocks
645  in the result is such that along every index the blocks are
646  organized according to rising qnum.
647 
648  Note that this means that the end result changes if the
649  directions of some of the legs are flipped before calling
650  to_ndarray(). Thus if for example trace or dot is called on the
651  resulting numpy array, the result may be different than for the
652  AbelianTensor if the contraction requires flipping directions.
653  Similarly taking for example traces and diags along axes that
654  were not compatible in the AbelianTensor is a perfectly valid
655  thing to do for the ndarray, and gives different results.
656 
657  All these concerns can be avoided by making sure that one only
658  calls on the ndarray operations that would have been valid on
659  the AbelianTensor without flipping any directions.
660  """
661  ndshape = type(self).flatten_shape(self.shape)
662  res = np.full(ndshape, self.defval, dtype=self.dtype)
663  if 0 in ndshape:
664  return res
665  shp, qhp = type(self).sorted_shape_qhape(tensor=self)
666  # ranges is like shape, but every number d is replaced by a
667  # tuple (a,a+d) where a is the sum of all the previous entries
668  # in the same dim.
669  ranges = []
670  for dim in shp:
671  prv = dim[0]
672  r = [(0, prv)]
673  for d in dim[1:]:
674  nxt = prv + d
675  r.append((prv, nxt))
676  prv = nxt
677  ranges.append(r)
678  for k, v in self.sects.items():
679  slc = ()
680  for i, qnum in enumerate(k):
681  r = ranges[i][qhp[i].index(qnum)]
682  slc += (slice(r[0], r[1]),)
683  res[slc] = v
684  return res

772  def flip_dir(self, axis):
773  """ Flips the direction of the given axis of self. The operation
774  is not in-place, but a view is returned. The quantum numbers
775  along given axis are also negated accordingly.
776  """
777  res = self.empty_like()
778  res.dirs[axis] *= -1
779  if self.qodulus is None:
780  qod_func = lambda q: q
781  else:
782  qod_func = lambda q: q % self.qodulus
783  res.qhape[axis] = [qod_func(-q) for q in res.qhape[axis]]
784  keys = set(self.sects.keys())
785  while keys:
786  k = keys.pop()
787  kf = list(k)
788  kf[axis] = qod_func(-kf[axis])
789  kf = tuple(kf)
790  vf = self[k]
791  if kf in keys:
792  v = self[kf]
793  keys.discard(kf)
794  res[k] = v
795  res[kf] = vf
796  return res

949  @classmethod
950  def find_trunc_dim(cls, S, S_sects, minusabs_next_els, dims,
951  chis=None, eps=0, break_degenerate=False,
952  degeneracy_eps=1e-6, norm_type="frobenius"):
953  # First, find what chi will be.
954  S = -np.sort(-np.abs(S))
955  if norm_type=="frobenius":
956  S = S**2
957  eps = eps**2
958  elif norm_type=="trace":
959  pass
960  else:
961  raise ValueError("Unknown norm_type {}".format(norm_type))
962  sum_all = np.sum(S)
963  if sum_all != 0:
964  for chi in chis:
965  if not break_degenerate:
966  # Make sure that we don't break degenerate singular
967  # values by including one but not the other.
968  while 0 < chi < len(S):
969  last_eig_in = S[chi-1]
970  last_eig_out = S[chi]
971  rel_diff = np.abs(last_eig_in-last_eig_out)/last_eig_in
972  if rel_diff < degeneracy_eps:
973  chi -= 1
974  else:
975  break
976  sum_disc = sum(S[chi:])
977  rel_err = sum_disc/sum_all
978  if rel_err <= eps:
979  break
980  if norm_type=="frobenius":
981  rel_err = np.sqrt(rel_err)
982  else:
983  rel_err = 0
984  chi = min(chis)
985  # Find out which eigenvalues to keep.
986  dim_sum = 0
987  while(dim_sum < chi):
988  # index_to_add is the index of the block which has
989  # the largest eigenvalue that is not yet
990  # included in truncation.
991  try:
992  minusabs_el_to_add, key = heapq.heappop(minusabs_next_els)
993  except IndexError:
994  # All the dimensions are fully included.
995  break
996  dims[key] += 1
997  this_key_els = S_sects[key][0]
998  if dims[key] < len(this_key_els):
999  next_el = this_key_els[dims[key]]
1000  heapq.heappush(minusabs_next_els, (-np.abs(next_el), key))
1001  dim_sum += 1
1002  return chi, dims, rel_err

1010  def join_indices(self, *inds, dirs=None,
1011  return_transposed_shape_data=False):
1012  """ Joins indices together in the spirit of reshape. inds is
1013  either a iterable of indices, in which case they are joined, or
1014  a iterable of iterables of indices, in which case the indices
1015  listed in each element of inds will be joined.
1016 
1017  Before any joining is done the indices are transposed so that
1018  for every batch of indices to be joined the first remains in
1019  place and the others are moved to be after in the order given.
1020  The order in which the batches are given does not matter.
1021 
1022  dirs are the directions of the new indices, defaults to
1023  [1,...,1]. If a batch of indices to be joined consists of only
1024  one index, its direction will be flipped to be as in dirs.
1025 
1026  If return_transposed_shape_data is True, then the shape, qhape
1027  and dirs (in this order) of the tensor after transposing but
1028  before reshaping are returned as well.
1029 
1030  The method does not modify the original tensor.
1031  """
1032  # Format index_batches to be a list of lists of indices.
1033  if isinstance(inds[0], collections.Iterable):
1034  index_batches = list(map(list, inds))
1035  else:
1036  index_batches = [list(inds)]
1037  # Remove empty batches.
1038  index_batches = [b for b in index_batches if len(b)>0]
1039 
1040  if dirs is None:
1041  warnings.warn("In join_indices, dirs was not given and is thus "
1042  "generated to be [1,...,1].")
1043  dirs = [1]*len(index_batches)
1044  else:
1045  if not isinstance(dirs, collections.Iterable):
1046  dirs = [dirs]
1047  assert(len(dirs) == len(index_batches))
1048 
1049  if not index_batches:
1050  # Nothing to be done. However, join_indices should always
1051  # return an array independent of the original, so we take a
1052  # view.
1053  if return_transposed_shape_data:
1054  return (self.view(), self.shape.copy(), self.qhape.copy(),
1055  self.dirs.copy())
1056  else:
1057  return self.view()
1058 
1059  # Group dirs together with index_batches so that they get sorted
1060  # together.
1061  index_batches_with_dirs = [b + [d] for b,d in zip(index_batches, dirs)]
1062 
1063  # Create the permutation for transposing the tensor. At the same
1064  # time transpose and sort index_batches.
1065  joined = set(sum(index_batches, []))
1066  not_joined = [[i] for i in range(len(self.shape)) if i not in joined]
1067  all_batches = not_joined + index_batches_with_dirs
1068  all_batches.sort(key=operator.itemgetter(0))
1069  # The a[:-1] conditional statement leaves out the dirs when
1070  # creating the permutation.
1071  p = sum((a[:-1] if len(a)>1 else a for a in all_batches), [])
1072  index_batches_with_dirs = [b for b in all_batches if len(b) > 1]
1073  dirs = [b[-1] for b in index_batches_with_dirs]
1074  index_batches = [b[:-1] for b in index_batches_with_dirs]
1075  index_batches = [list(map(p.index, b)) for b in index_batches]
1076  res = self.transpose(p)
1077 
1078  if return_transposed_shape_data:
1079  transposed_shape = res.shape.copy()
1080  transposed_qhape = res.qhape.copy()
1081  transposed_dirs = res.dirs.copy()
1082 
1083  # Flip the dirs on single indices in index_batches and remove
1084  # those batches from the list.
1085  for i,b in reversed(tuple(enumerate(index_batches))):
1086  if len(b) == 1:
1087  if res.dirs[b[0]] != dirs[i]:
1088  res = res.flip_dir(b[0])
1089  del(dirs[i])
1090  del(index_batches[i])
1091 
1092  if not index_batches:
1093  # If no indices are left, there is no need to join anything.
1094  if return_transposed_shape_data:
1095  return res, transposed_shape, transposed_qhape, transposed_dirs
1096  else:
1097  return res
1098 
1099  # Find out the remaining, new indices after the joining.
1100  cumulant = 0
1101  new_inds = []
1102  for b in index_batches:
1103  new_inds.append(b[0] - cumulant)
1104  cumulant += len(b) - 1
1105 
1106  # Reverse index_batches and dirs for the future so that we first
1107  # process the indices at the end.
1108  index_batches.reverse()
1109  dirs.reverse()
1110 
1111  # For every non-zero block in res, reshape the block and add it
1112  # to the right key in new_sects. However, every item in
1113  # new_sects will consist of several blocks that need to be
1114  # concatenated. Because of this, new_sects[k] is a list of
1115  # lists [k_part1, k_part2, ..., k_partn, reshaped_block],
1116  # where k_parts are the qnums of the indices that were joined.
1117  # Thus by later sorting these lists we get them in the right
1118  # order for concatenation.
1119  new_sects = {}
1120  if res.qodulus is None:
1121  qod_func = lambda x: x
1122  else:
1123  qod_func = lambda x: x % res.qodulus
1124  # Go through every valid index in stead of every key in sects,
1125  # because blocks of zeros may be concatenated with other blocks.
1126  valid_ks = (qcomb for qcomb in itertools.product(*res.qhape)
1127  if res.is_valid_key(qcomb))
1128  del_slcs = [slice(b[1], b[-1]+1) for b in index_batches]
1129  get_slcs = [slice(b[0], b[-1]+1) for b in index_batches]
1130  dir_batches = [[res.dirs[i] for i in batch] for batch in index_batches]
1131  for k in valid_ks:
1132  v = res[k]
1133  new_k = list(k)
1134  new_shp = list(v.shape)
1135  k_parts = []
1136  for b, dir_b, dir_new, del_slc, get_slc in\
1137  zip(index_batches, dir_batches, dirs, del_slcs, get_slcs):
1138  k_part = k[get_slc]
1139  k_parts.append(k_part)
1140  k_part = map(operator.mul, k_part, dir_b)
1141  new_qnum = qod_func(sum(k_part) * dir_new)
1142  new_k[b[0]] = new_qnum
1143  del(new_k[del_slc])
1144  new_shp[b[0]] = reduce(operator.mul, v.shape[get_slc])
1145  del(new_shp[del_slc])
1146  k_parts.reverse()
1147  new_k = tuple(new_k)
1148  l = new_sects.setdefault(new_k, [])
1149  l.append(k_parts + [v.reshape(new_shp)])
1150 
1151  # Concatenator is a helper function that recursively
1152  # concatenates the pieces together. It is called once for every
1153  # index in a batch.
1154  def concatenator(l, i=0):
1155  if i == len(l[0])-2:
1156  l = [el[-1] for el in l]
1157  else:
1158  l = [tuple(g)
1159  for k,g in itertools.groupby(l, operator.itemgetter(i))]
1160  l = tuple(map(lambda k: concatenator(k, i=i+1), l))
1161  return np.concatenate(l, new_inds[i])
1162

1164  # These are the new blocks, just need to concatenate.
1165  v.sort()
1166  new_sects[k] = concatenator(v)
1167  res.sects = new_sects

1213  def split_indices(self, indices, dims, qims=None, dirs=None):
1214  """ Splits indices in the spirit of reshape. Indices is an
1215  iterable of indices to be split. Dims is an iterable of
1216  iterables such that dims[i]=dim_batch is an iterable of lists of
1217  dimensions, each list giving the dimensions along a new index
1218  that will come out of splitting indices[i]. qims correspondingly
1219  gives the qims of the new indices, and dirs gives the new
1220  directions.
1221 
1222  An example clarifies:
1223  Suppose self has shape [dim1, dim2, dim3, dim4], qhape [qim1,
1224  qim2, qim3, qim4] and dirs [d1,d2,d3,d4]. Suppose then that
1225  indices = [1,3], dims = [[dimA, dimB], [dimC, dimD]], qims =
1226  [[qimA, qimB], [qimC, qimD]] and dirs = [[dA, dB] [dC, dD]].
1227  Then the resulting tensor will have shape [dim1, dimA, dimB,
1228  dim3, dimC, dimD], qhape [qim1, qimA, qimB, qim3, qimC, qimD]
1229  and dirs [d1, dA, dB, d3, dC, dD]. All this assuming that that dims
1230  and qims are such that joining qimA and qimB gives qim2, etc.
1231 
1232  In stead of a list of indices a single index may be given.
1233  Correspondingly dims, qims and dirs should then have one level
1234  of depth less as well.
1235 
1236  split_indices never modifies the original tensor.
1237  """
1238  # Formatting the input so that indices is a list and dim_batches
1239  # and dim_batches are lists of lists.
1240  if isinstance(indices, collections.Iterable):
1241  assert(len(indices) == len(dims) == len(qims))
1242  indices = list(indices)
1243  dim_batches = list(map(list, dims))
1244  qim_batches = list(map(list, qims))
1245  if dirs is None:
1246  warnings.warn("In split_indices, dirs was not given and is "
1247  "thus generated to be full of ones.")
1248  dir_batches = list(map(lambda dim: [1]*len(dim), dims))
1249  else:
1250  dir_batches = dirs
1251  else:
1252  indices = [indices]
1253  dim_batches = [list(dims)]
1254  qim_batches = [list(qims)]
1255  if dirs is None:
1256  warnings.warn("In split_indices, dirs was not given and is "
1257  "thus generated to be [[1,...,1]].")
1258  dir_batches = [[1]*len(dims)]
1259  else:
1260  dir_batches = [dirs]
1261 
1262  if not indices:
1263  return self.view()
1264 
1265  # Reverse sort according to indices, so that we always handle
1266  # the last index first. This way when removing elements from
1267  # different iterables we always remove from the end and don't
1268  # mess up the indexing.
1269  indices, dim_batches, qim_batches, dir_batches =\
1270  zip(*sorted(zip(indices, dim_batches, qim_batches, dir_batches),
1271  reverse=True))
1272 
1273  if self.qodulus is None:
1274  qod_func = lambda x: x
1275  else:
1276  qod_func = lambda x: x % self.qodulus
1277 
1278  # Step one: Build split_data. It will be a dictionary with keys
1279  # (ind, qnum), so that
1280  # split_data[ind, qnum] = [(qcomb1, dcomb1, dcum1),
1281  # (qcomb2, dcomb2, dcum2),
1282  # ... ]
1283  # Each of these tuples 1,2, ... will correspond to a new block
1284  # in the split tensor. Every qcomb is a tuple of quantum
1285  # numbers, one for each new index coming out of ind, such that
1286  # their combination is qnum. Every dcomb is a tuple of
1287  # corresponding dimensions. Every dcum is a tuple of two numbers
1288  # (a,b) such that when slicing a block of self, the slice along
1289  # ind to that gets the block in question in dcum[0]:dcum[1].
1290  split_data = {}
1291  for ind, dims, qims, dir_batch in zip(indices, dim_batches,
1292  qim_batches, dir_batches):
1293  # Find combinations of qims that give the right qnum
1294  dimcombs = itertools.product(*dims)
1295  qimcombs = itertools.product(*qims)
1296  for qcomb, dcomb in zip(qimcombs, dimcombs):
1297  qcomb_flipped = tuple(map(operator.mul, qcomb, dir_batch))
1298  qnum = qod_func(sum(qcomb_flipped)*self.dirs[ind])
1299  if self.qodulus is not None:
1300  qnum %= self.qodulus
1301  if qnum in self.qhape[ind]:
1302  try:
1303  split_data[ind, qnum].add((qcomb, dcomb))
1304  except KeyError:
1305  split_data[ind, qnum] = {(qcomb, dcomb)}
1306 
1307  # Sort split_data items and add dcums.
1308  for k,v in split_data.items():
1309  v = sorted(v)
1310  cumulant = 0
1311  for i, el in enumerate(v):
1312  prod = reduce(operator.mul, el[1])
1313  v[i] += ((cumulant, cumulant + prod),)
1314  cumulant += prod
1315  split_data[k] = v
1316 
1317  # Step two: Use split_data to do the splitting.
1318  new_sects = {}
1319  for k,v in self.sects.items():
1320  # For every block v that is to be split, we need the
1321  # corresponding splitting data for every index that is being
1322  # split.
1323  datas = map(lambda i: split_data[i, k[i]], indices)
1324  # We then take the cartesian product of these datas over the
1325  # indices being split. That is, every member of the product
1326  # corresponds to one of the indices.
1327  for p in itertools.product(*datas):
1328  # p is a tuple of lists, one list for each ind, where
1329  # each list consists of three tuples: a qcomb, a dcomb
1330  # and a cumdim. For each p there will be a block in the
1331  # new tensor.
1332  #
1333  # Build the key for the new block, the slice that gets
1334  # the new block from v, and the shape that the new block
1335  # shall be reshaped to.
1336  s = slice(None)
1337  slc = [s]*len(self.shape)
1338  new_key = list(k)
1339  block_shape = list(v.shape)
1340  for i, t in enumerate(p):
1341  qcomb = t[0]
1342  dcomb = t[1]
1343  cumdim = t[2]
1344  new_key[indices[i]:indices[i]+1] = qcomb
1345  block_shape[indices[i]:indices[i]+1] = dcomb
1346  slc[indices[i]] = slice(cumdim[0], cumdim[1])
1347  new_key = tuple(new_key)
1348  # Blam!
1349  new_sects[new_key] = v[slc].reshape(block_shape)
1350  res = self.empty_like()
1351  res.sects = new_sects
1352 
1353  # Create the new shape, qhape and dirs.
1354  for ind, dim_b, qim_b, dir_b in zip(indices, dim_batches, qim_batches,
1355  dir_batches):
1356  res.shape[ind:ind+1] = dim_b
1357  res.qhape[ind:ind+1] = qim_b
1358  res.dirs[ind:ind+1] = dir_b
1359 
1360  return res

1397  def trace(self, axis1=0, axis2=1):
1398  """ Takes a trace of self over axis1 and axis2. This differs
1399  from the usual trace in the sense that it is more like
1400  connecting the two legs and contracting. This means that if the
1401  legs axis1 and axis2 don't have the same dim and qim the
1402  function will raise an error. If the dirs don't match (both are
1403  1 or both are -1) then one of them is flipped and a warning is
1404  raised.
1405 
1406  Note that the diagonal consists always of blocks with the same
1407  qnum on axis1 and axis2 (once dirs are opposite). This means
1408  that the trace of an invariant charge != 0 tensor is always a
1409  zero-tensor.
1410  """
1411  assert(self.compatible_indices(self, axis1, axis2))
1412  if axis1 < axis2:
1413  axis1, axis2 = axis2, axis1
1414  if self.dirs[axis1] + self.dirs[axis2] != 0:
1415  warnings.warn("Automatically flipping dir %i in trace."%axis1,
1416  stacklevel=2)
1417  self = self.flip_dir(axis1)
1418  res = self.empty_like()
1419  del(res.shape[axis1])
1420  del(res.shape[axis2])
1421  del(res.qhape[axis1])
1422  del(res.qhape[axis2])
1423  del(res.dirs[axis1])
1424  del(res.dirs[axis2])
1425 
1426  # We could avoid looping over the whole dictionary by
1427  # constructing the cartesiand product of all the qims but not
1428  # qhape[axis2], and then infering what the qnum on axis2 should
1429  # be. I suspect that this would be slower, but I haven't tried.
1430  for k,v in self.sects.items():
1431  qnum = k[axis1] - k[axis2]
1432  if self.qodulus is not None:
1433  qnum %= self.qodulus
1434  if qnum == 0:
1435  new_k = tuple(i for j,i in enumerate(k)
1436  if j != axis1 and j != axis2)
1437  if new_k in res.sects:
1438  res[new_k] += v.trace(axis1=axis1, axis2=axis2)
1439  else:
1440  res[new_k] = v.trace(axis1=axis1, axis2=axis2)
1441  if not res.shape:
1442  # The result is a scalar.
1443  try:
1444  res.defval = res[()]
1445  res.sects = {}
1446  except KeyError:
1447  # There was no () block, so the tensor is 0 by symmetry.
1448  pass
1449  return res

1498  def matrix_dot(self, other):
1499  """ Takes the dot product of two tensors of rank < 3. If self or
1500  other is a matrix, it must be invariant and have defval == 0.
1501  """
1502  assert(self.qodulus == other.qodulus)
1503 
1504  # The following essentially a massive case statement on whether
1505  # self and other are scalar, vectors or matrices. Unwieldly, but
1506  # efficient and clear.
1507  if not self.shape and not other.shape:
1508  return self*other
1509  else:
1510  res_dtype = np.result_type(self.dtype, other.dtype)
1511  if self.qodulus is None:
1512  qod_func = lambda x: x
1513  else:
1514  qod_func = lambda x: x % self.qodulus
1515  res_charge = qod_func(self.charge + other.charge)
1516  res_invar = self.invar and other.invar
1517 
1518  # Vector times vector
1519  if len(self.shape) == 1 and len(other.shape) == 1:
1520  assert(self.compatible_indices(other, 0, 0))
1521  if self.dirs[0] + other.dirs[0] != 0:
1522  warnings.warn("Automatically flipping dir 0 of other in "
1523  "dot.")
1524  other = other.flip_dir(0)
1525  res = 0
1526  for qnum in self.qhape[0]:
1527  try:
1528  a = self[(qnum,)]
1529  b = other[(qnum,)]
1530  except KeyError:
1531  continue

1533  if prod:
1534  res += np.dot(a,b)
1535  res = type(self)([], qhape=[], qodulus=self.qodulus, sects={},
1536  defval=res, dirs=[], dtype=res_dtype,
1537  charge=res_charge, invar=res_invar)
1538  else:
1539  res_sects = {}
1540 
1541  # Vector times matrix
1542  if len(self.shape) == 1:
1543  assert(other.invar)
1544  assert(other.defval == 0)
1545  assert(self.compatible_indices(other, 0, 0))

1548  "in dot.")

1625  def matrix_eig(self, chis=None, eps=0, print_errors=0, hermitian=False,
1626  break_degenerate=False, degeneracy_eps=1e-6,
1627  norm_type="frobenius"):
1628  """ Find eigenvalues and eigenvectors of a matrix. The input
1629  must have defval == 0, invar == True, charge == 0 and must be
1630  square in the sense that the dimensions must have the same qim
1631  and dim and opposing dirs.
1632 
1633  If hermitian is True the matrix is assumed to be hermitian.
1634 
1635  Truncation works like for SVD.
1636 
1637  The output is in the form S, U, where S is a non-invariant
1638  vector of eigenvalues and U is a matrix that has its columns the
1639  eigenvectors. Both have the same dim and qim as self.
1640  """
1641  chis = self.matrix_decomp_format_chis(chis, eps)
1642  np_func = np.linalg.eigh if hermitian else np.linalg.eig
1643  assert(self.defval == 0)
1644  assert(self.invar)
1645  assert(self.charge == 0)
1646  assert(self.dirs[0] + self.dirs[1] == 0)
1647  assert(set(zip(self.qhape[0], self.shape[0])) ==
1648  set(zip(self.qhape[1], self.shape[1])))
1649  if self.qodulus is None:
1650  qod_func = lambda x: x
1651  else:
1652  qod_func = lambda x: x % self.qodulus
1653 
1654  S_dtype = np.float_ if hermitian else np.complex_
1655  U_dtype = self.dtype if hermitian else np.complex_
1656 
1657  eigdecomps = {}
1658  dims = {}
1659  minusabs_next_eigs = []
1660  all_eigs = []
1661  for k,v in self.sects.items():
1662  if 0 not in v.shape:
1663  s, u = np_func(v)
1664  order = np.argsort(-np.abs(s))
1665  s = s[order]
1666  u = u[:,order]
1667  s = s.astype(S_dtype)
1668  u = u.astype(U_dtype)
1669  eigdecomp = (s, u)
1670  else:
1671  shp = v.shape
1672  m = min(shp)
1673  u = np.empty((shp[0], m), dtype=U_dtype)
1674  s = np.empty((m,), dtype=S_dtype)
1675  eigdecomp = (u, s)
1676  eigdecomps[k] = eigdecomp
1677  dims[k] = 0
1678  all_eigs.append(s)
1679  if 0 not in s.shape:
1680  heapq.heappush(minusabs_next_eigs, (-np.abs(s[0]), k))
1681  try:
1682  all_eigs = np.concatenate(all_eigs)
1683  except ValueError:
1684  # all_eigs == []
1685  all_eigs = np.array((0,))
1686 
1687  # Truncate, if truncation dimensions are given.
1688  chi, dims, rel_err = type(self).find_trunc_dim(
