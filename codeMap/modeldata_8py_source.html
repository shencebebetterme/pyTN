145 def build_primas(prima_pairs):
146  """ Given a list of pairs of holomorphic and anti-holomorphic parts
147  of primaries, construct the scaling dimensions, spins and
148  degeneracies of the towers of these primaries.
149  """
150  scaldims = []
151  spins = []
152  degs = []
153  for left, right in prima_pairs:
154  # Virs refer to Virasoro characters.
155  h, Virs_h = left
156  hbar, Virs_hbar = right
157  i_Virs_h = enumerate(Virs_h)
158  i_Virs_hbar = enumerate(Virs_hbar)
159  for (i, Vir_h), (j, Vir_hbar) in it.product(i_Virs_h, i_Virs_hbar):
160  deg = Vir_h*Vir_hbar
161  if deg > 0:
162  scaldims.append(h+i + hbar+j)
163  spins.append(h+i - hbar-j)
164  degs.append(deg)
165  # We make sure that values that are the same up to numerical errors
166  # are stored as exactly the same. This is necessary later, when
167  # sorting is done on these values.
168  for li in (scaldims, spins, degs):
169  past = set()
170  for i, el in enumerate(li):
171  for old_el in past:
172  if abs(el-old_el) < 1e-7:
173  # The elements are considered the same and made
174  # exactly equal.
175  el = old_el
176  break
177  past.add(el)
178  li[i] = el
179  return scaldims, spins, degs
