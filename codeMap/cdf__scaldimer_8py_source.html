114 def generate_initial_tensors(pars):
115  ham = (- pars["J"]*np.array([[ 1,-1],
116  [-1, 1]],
117  dtype=pars["dtype"])
118  + pars["H"]*np.array([[-1, 0],
119  [ 0, 1]],
120  dtype=pars["dtype"]))
121  boltz = np.exp(-pars["beta"]*ham)
122  ham_g = - pars["g"]* pars["J"]*np.array([[ 1,-1],
123  [-1, 1]],
124  dtype=pars["dtype"])
125  boltz_g = np.exp(-pars["beta"]*ham_g)
126  ones = np.ones((2,2), dtype=pars["dtype"])
127  D = np.einsum('ab,bc,cd,da->abcd', boltz, boltz_g, boltz_g, boltz)
128  if pars["symmetry_tensors"]:
129  u = np.array([[1, 1],
130  [1,-1]]) / np.sqrt(2)
131  u_dg = u.T.conjugate()
132  D = scon((D, u, u, u_dg, u_dg),
133  ([1,2,3,4], [-1,1], [-2,2], [3,-3], [4,-4]))

260  # Use iterative diagonalization.
261  block_width = pars["block_width"]

265  try:
266  qimT = T.qhape[3]
267  except TypeError:
268  qimT = None
269  dim1 = A1.shape[3]
270  try:

278  qim2 = None

280  flatdim1 = type(T).flatten_dim(dim1)

282  matrix_flatdim = flatdimT**(block_width-2)*flatdim1*flatdim2

284  def caster(v, charge=0):
285  v = type(T).from_ndarray(v, shape=[dim1,dim2] + [dimT]*(block_width-2),
286  qhape=[qim1,qim2] + [qimT]*(block_width-2),
287  charge=charge, dirs=[1]*block_width)
288  return v
289 
290  scon_list_end = [A1, A2] + [T]*(block_width-2)
291  index_list = [[i*2 + 1 for i in range(block_width)],
292  [block_width*2, -1, 2, 1], [2, -2, 4, 3]]
293  for i in range(3, block_width+1):
294  index_list += [[2*i-2, -i, 2*i, 2*i-1]]
295 
296  def transfer_op(v, charge=0):
297  v = np.reshape(v, (flatdim1, flatdim2) + (flatdimT,)*(block_width-2))
298  v = caster(v, charge=charge)
299  print(".", end='', flush=True)
300  scon_list = [v] + scon_list_end
301  Av = scon(scon_list, index_list)
302  Av = Av.to_ndarray()
303  Av = np.reshape(Av, (matrix_flatdim,))
304  return Av
305 
306  print("Diagonalizing...", end="")
307  if hasattr(T, "sects"):
308  qnums = [0,1]
309  es = type(T).empty(shape=[[n_dims_do]*len(qnums)],
