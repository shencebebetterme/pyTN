109 def connect_graph(AA, v, order):
110  """ Connect the graph of tensors to be contracted by trivial
111  indices, if necessary. Add these trivial indices to the end of the
112  contraction order.
113 
114  AA, v and order are modified in place.
115  """
116  # Build ccomponents, a list of the connected components of the graph,
117  # where each component is represented by a a set of indices.
118  unvisited = set(range(len(AA)))
119  visited = set()
120  ccomponents = []
121  while unvisited:
122  component = set()
123  next_visit = unvisited.pop()
124  to_visit = {next_visit}
125  while to_visit:
126  i = to_visit.pop()
127  unvisited.discard(i)
128  component.add(i)
129  visited.add(i)
130  # Get the indices of tensors neighbouring AA[i].
131  i_inds = set(v[i])
132  neighs = (j for j, j_inds in enumerate(v) if i_inds.intersection(j_inds))
133  for neigh in neighs:
134  if neigh not in visited:
135  to_visit.add(neigh)
136  ccomponents.append(component)
137  # If there is more than one connected component, take one of them, a
138  # take an arbitrary tensor (called c) out of it, and connect that
139  # tensor with an arbitrary tensor (called d) from all the other
140  # components using a trivial index.
141  c = ccomponents.pop().pop()
142  while ccomponents:
143  d = ccomponents.pop().pop()
144  A_c = AA[c]
145  A_d = AA[d]
146  c_axis = len(v[c])
147  d_axis = len(v[d])
148  try:
149  AA[c] = A_c.expand_dims(c_axis, direction=1)
150  except AttributeError:
151  AA[c] = np.expand_dims(A_c, c_axis)
152  try:
153  AA[d] = A_d.expand_dims(d_axis, direction=-1)
154  except AttributeError:
155  AA[d] = np.expand_dims(A_d, d_axis)
156  try:
157  dim_num = max(order)+1
158  except ValueError:
159  dim_num = 1
160  v[c].append(dim_num)
161  v[d].append(dim_num)
162  order.append(dim_num)
163  return None

248  """ Check that
249  1) the number of tensors in AA matches the number of index lists in v.
250  2) every tensor is given the right number of indices.

252  exactly once.

254  """

267  raise ValueError(('In scon.do_check_indices, len(v[%i])=%i '
